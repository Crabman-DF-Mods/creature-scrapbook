c_variation_misc

[OBJECT:CREATURE_VARIATION]
Must save at least 29 characters + the amount in the variation name,
to be worth making a template.

// This is for when you use COPY_TAGS_FROM, to remove tags that shouldn't be
// inherited from other creatures pretty much ever. You should generally always
// be redefining these tokens from scratch every time.
[CREATURE_VARIATION:REMOVE_IDENTITY]
	[CV_REMOVE_TAG:DESCRIPTION]
	[CV_REMOVE_TAG:NAME]
	[CV_REMOVE_TAG:GENERAL_CHILD_NAME][CV_REMOVE_TAG:GENERAL_BABY_NAME]

	// The tiles can sometimes be copied, but they're easy enough to replace anyway.
	[CV_REMOVE_TAG:CREATURE_TILE][CV_REMOVE_TAG:COLOR][CV_REMOVE_TAG:ALTTILE]
	[CV_REMOVE_TAG:GLOWTILE][CV_REMOVE_TAG:GLOWCOLOR]
	[CV_REMOVE_TAG:CREATURE_SOLDIER_TILE][CV_REMOVE_TAG:CREATURE_SOLDIER_ALTTILE]

	// All the same stuff as above, but specific to castes.
	[CV_REMOVE_TAG:CASTE_NAME]
	[CV_REMOVE_TAG:CHILDNAME][CV_REMOVE_TAG:BABYNAME]

	[CV_REMOVE_TAG:CASTE_TILE][CV_REMOVE_TAG:CASTE_COLOR][CV_REMOVE_TAG:CASTE_ALTTILE]
	[CV_REMOVE_TAG:CASTE_GLOWTILE][CV_REMOVE_TAG:CASTE_GLOWCOLOR]
	[CV_REMOVE_TAG:CASTE_SOLDIER_TILE][CV_REMOVE_TAG:CASTE_SOLDIER_ALTTILE]

	// Mundane (real life) creatures should never be created using COPY_TAGS_FROM.
	[CV_REMOVE_TAG:MUNDANE]

// Set the rarity data of a creature, either for the first time, or to overwrite
// it if you're modding another creature or using COPY_TAGS_FROM. You can use DEFAULT
// instead to simply remove all rarity-related tokens.
//
// Remember to use the '|' symbol in the args for POPULATION_NUMBER and CLUSTER_NUMBER.
[CREATURE_VARIATION:SET_RARITY]
	[CV_REMOVE_TAG:UBIQUITOUS]
	[CV_NEW_TAG:POPULATION_NUMBER:!ARG1]
	[CV_NEW_TAG:CLUSTER_NUMBER:!ARG2]
	[CV_NEW_TAG:FREQUENCY:!ARG3]

	[CV_REMOVE_CTAG:1:DEFAULT:POPULATION_NUMBER]	// 1:1
	[CV_REMOVE_CTAG:1:DEFAULT:CLUSTER_NUMBER]		// 1:1
	[CV_REMOVE_CTAG:1:DEFAULT:FREQUENCY]			// I don't know the default for this
